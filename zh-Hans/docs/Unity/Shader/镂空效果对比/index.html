Shader "Custom/CircleHole"
{
    Properties
    {
        _MainTex ("Texture", 2D) = "white" {}
        _Center ("Hole Center", Vector) = (0.5, 0.5, 0, 0)
        _Radius ("Hole Radius", Range(0, 0.5)) = 0.3
        _SoftEdge ("Edge Softness", Range(0, 0.1)) = 0.01
    }
    
    SubShader
    {
        Tags { "RenderType"="Opaque" }
        
        Pass
        {
            CGPROGRAM
            #pragma vertex vert
            #pragma fragment frag
            #include "UnityCG.cginc"

            struct appdata
            {
                float4 vertex : POSITION;
                float2 uv : TEXCOORD0;
            };

            struct v2f
            {
                float4 vertex : SV_POSITION;
                float2 uv : TEXCOORD0;
            };

            sampler2D _MainTex;
            float2 _Center;
            float _Radius;
            float _SoftEdge;

            v2f vert (appdata v)
            {
                v2f o;
                o.vertex = UnityObjectToClipPos(v.vertex);
                o.uv = v.uv;
                return o;
            }
