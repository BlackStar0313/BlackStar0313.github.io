Shader纹理保持亮度渐变颜色

效果​

需求​

  • 转盘游戏，游戏中有个蓝色的背景，美术希望在得到大奖后，背景颜色从蓝色渐变到金色。
  • 直接修改图片的颜色，打不到效果，而且希望渐变式的改变到对应颜色。

这里是美术出的效果图 蓝色原始图和金色渐变图

实现​

思路为：

如果直接在片元着色器中，采样原始纹理，直接给原像素点添加一个色值，会导致纹理叠颜色后整个糊到一起了，感觉想整个图片上蒙了一层颜色。

正确的应该是保持原纹理中像素点的亮度，然后改变色相和饱和度。这样看起来背景里画的内容都和原图一样保持清晰，整体饱和度色相变成了别的颜色。

最后由于目标颜色更亮一点，最后再找一个参数让整体亮度再亮一点，能有更暖的效果。

具体操作​

1. 写采样用的shader​

TintOverlay.effect
// TintOverlay.effect - 渐变过程式的叠加颜色效果
CCEffect %{
techniques:
- passes:
- vert: vs
  frag: fs
  blendState:
    targets:
    - blend: true
  rasterizerState:
    cullMode: none
  properties:
    texture: { value: white }
    overlayColor: { value: [1.0, 0.6, 0.0, 1.0], editor: { type: color } }
    progress: { value: 0.0 }
    brightness: { value: 1.25 }   // 新增
}%

CCProgram vs %{
precision highp float;
#include <cc-global>
#include <cc-local>

in vec3 a_position;
in vec2 a_uv0;
in vec4 a_color;

out vec2 v_uv0;
out vec4 v_color;

void main () {
vec4 pos = vec4(a_position, 1.0);
#if CC_USE_MODEL
  pos = cc_matViewProj * cc_matWorld * pos;
#else
  pos = cc_matViewProj * pos;
#endif
v_uv0 = a_uv0;
v_color = a_color;
gl_Position = pos;
}
}%

CCProgram fs %{
precision highp float;
#include <alpha-test>

in vec2 v_uv0;
in vec4 v_color;
uniform sampler2D texture;

uniform ARGS {
vec4 overlayColor;
float progress;
float brightness;   // 新增
float _pad0;
};

void main () {
vec4 base = texture(texture, v_uv0);
float k = clamp(progress, 0.0, 1.0) * clamp(overlayColor.a, 0.0, 1.0);

// 保留原图亮度（luminance），使用叠加颜色的色相/饱和度
const vec3 LUMA = vec3(0.2126, 0.7152, 0.0722);
float L  = dot(base.rgb, LUMA);
float Lo = max(dot(overlayColor.rgb, LUMA), 1e-4);

vec3 colorized = overlayColor.rgb * (L / Lo) * brightness; // 保亮度并增益
// 在原图与“保亮度的着色”之间插值
vec3 rgb = mix(base.rgb, clamp(colorized, 0.0, 1.0), k);
gl_FragColor = vec4(rgb, base.a) * v_color;
}
}%

4. 使用脚本控制​

这里的渐变过程，我用tween来控制，tween的进度值，就是shader中的progress值。具体目标颜色，不能直接吸取目标图片里的颜色，需要来回挑一挑看效果，确定一个合适的颜色。

TintOverlay.ts
export default class SpriteTintOverlayEffect extends cc.Component {
@property(cc.Sprite)
bgSprite: cc.Sprite = null;

private material: cc.Material = null;
private tweenCarrier: { t: number } = null;
private activeTween: cc.Tween<{ t: number }>;

async onLoad() {
	await this.useTintMaterial();
}

private async useTintMaterial() {
	// 获取材质
	if (!this.material && this.bgSprite) {
		this.material = this.bgSprite.getMaterial(0);
		console.log("Got material from sprite:", this.material);
	}
}

onDestroy(): void {
	this.stopTintTween();
}

onDisable() {
	this.stopTintTween();
}

private stopTintTween() {
	if (this.activeTween) {
		try { this.activeTween.stop(); } catch (e) { }
		this.activeTween = null;
	}
	if (this.tweenCarrier) {
		try { cc.Tween.stopAllByTarget(this.tweenCarrier); } catch (e) { }
		this.tweenCarrier = null;
	}
	if (this.material) {
		// 复位效果，避免残留
		try { this.material.setProperty('progress', 0); } catch (e) { }
	}
}

// 开始一次“0.3s -> 停 5s -> 0.3s”叠色动画
playTint(color: cc.Color, staySec: number = 5, fadeSec: number = 0.3, brightness: number = 1.38) {
	if (!this.material) return;

	// 设置叠加颜色（把 cc.Color 转为 0~1）
	const oc = [
		color.r / 255,
		color.g / 255,
		color.b / 255,
		color.a / 255,  // 用 A 当强度
	];
	this.material.setProperty('overlayColor', oc);
	this.material.setProperty('brightness', brightness);

	this.stopTintTween();
	this.tweenCarrier = { t: 0 };
	this.material.setProperty('progress', 0);

	this.activeTween = cc.tween(this.tweenCarrier)
		.to(fadeSec, { t: 1 }, {
			progress: (s, e, c, r) => {
				const v = s + (e - s) * r;
				this.material.setProperty('progress', v);
				return v;
			},
		})
		.delay(staySec)
		.to(fadeSec, { t: 0 }, {
			progress: (s, e, c, r) => {
				const v = s + (e - s) * r;
				this.material.setProperty('progress', v);
				return v;
			},
		})
		.call(() => { this.activeTween = null; })
		.start();
}
}
