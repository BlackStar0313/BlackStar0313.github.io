Shader金色描边

效果​

需求​

  • 有一些特殊价值的鱼需要通过金色描边来表现。
  • 同样的一张图，唯一区别是有金色边和没金色边。可以让美术出两套图，由于游戏中鱼的种类有几十种，两套图只有描边不一样，会导致没必要的包体过大，渲染时候drawcall的也是两张纹理。所以对一套图，用shader来实现。

实现​

思路为：

描边效果，需要找到纹理中边缘像素进行颜色填充，通过对原图像素点alpha值判断，并对周围像素点进行采样。

若有像素点的alpha值为0，找周围像素点，若周围像素点alpha值不为0，则认为该像素点是边缘，需要上色。如果周围像素点采样后都为0，再往外扩大一圈，如果不为0，则认为该像素点是边缘，需要上色。这样可以上不同程度的颜色达到描边变色的效果。如果需要细腻一点重复上述过程。

关于边缘检测还有很多采样的方法，我这里选择了简单的8方向采样像素。

具体操作​

1. 写采样用的shader​

GoldOutline.effect
// GoldOutline.effect - 金色渐变描边效果
CCEffect %{
techniques:
- passes:
- vert: vs
  frag: fs
  blendState:
    targets:
    - blend: true
  rasterizerState:
    cullMode: none
  properties:
    texture: { value: white }
    outlineWidth: { 
      value: 2.0,
      editor: {
        tooltip: "描边宽度"
      }
    }
    outlineColor1: {
      value: [1.0, 0.8, 0.0, 1.0],
      editor: {
        type: color,
        tooltip: "描边内侧颜色（金色）"
      }
    }
    outlineColor2: {
      value: [1.0, 0.6, 0.0, 1.0],
      editor: {
        type: color,
        tooltip: "描边外侧颜色（橙金色）"
      }
    }
    outlineIntensity: {
      value: 1.0,
      editor: {
        tooltip: "描边强度"
      }
    }
    glowIntensity: {
      value: 0.5,
      editor: {
        tooltip: "发光强度"
      }
    }
    texSize: {
      value: [256.0, 256.0],
      editor: {
        tooltip: "纹理尺寸"
      }
    }
}%

CCProgram vs %{
precision highp float;
#include <cc-global>
#include <cc-local>

in vec3 a_position;
in vec2 a_uv0;
in vec4 a_color;

out vec2 v_uv0;
out vec4 v_color;

void main() {
vec4 pos = vec4(a_position, 1);

#if CC_USE_MODEL
pos = cc_matViewProj * cc_matWorld * pos;
#else
pos = cc_matViewProj * pos;
#endif

v_uv0 = a_uv0;
v_color = a_color;

gl_Position = pos;
}
}%

CCProgram fs %{
precision highp float;
#include <alpha-test>

in vec2 v_uv0;
in vec4 v_color;

uniform sampler2D texture;

uniform ARGS {
vec4 outlineColor1;     
vec4 outlineColor2;     
vec2 texSize;           // 纹理尺寸
float outlineWidth;     
float outlineIntensity; 
float glowIntensity;    
float padding;          
};

void main() {
vec4 originalColor = texture(texture, v_uv0);
vec2 texelSize = 1.0 / texSize;

// 计算描边
float outline = 0.0;

// 如果当前像素是透明的，检查周围是否有不透明像素
if (originalColor.a < 0.1) {
  // 8方向采样
  vec2 offsets[8];
  offsets[0] = vec2(-1.0, -1.0);
  offsets[1] = vec2( 0.0, -1.0);
  offsets[2] = vec2( 1.0, -1.0);
  offsets[3] = vec2(-1.0,  0.0);
  offsets[4] = vec2( 1.0,  0.0);
  offsets[5] = vec2(-1.0,  1.0);
  offsets[6] = vec2( 0.0,  1.0);
  offsets[7] = vec2( 1.0,  1.0);
  
  for (int i = 0; i < 8; i++) {
    vec2 samplePos = v_uv0 + offsets[i] * texelSize * outlineWidth;
    float alpha = texture(texture, samplePos).a;
    if (alpha > 0.1) {
      outline = 1.0;
      break;
    }
  }
  
  // 如果第一圈没找到，扩大搜索范围
  if (outline == 0.0 && outlineWidth > 1.0) {
    for (int i = 0; i < 8; i++) {
      vec2 samplePos = v_uv0 + offsets[i] * texelSize * outlineWidth * 1.5;
      float alpha = texture(texture, samplePos).a;
      if (alpha > 0.1) {
        outline = 0.6;
        break;
      }
    }
  }
}

// 输出结果
if (originalColor.a > 0.1) {
  // 原始像素，直接输出
  gl_FragColor = originalColor * v_color;
} else if (outline > 0.0) {
  // 描边像素
  vec4 finalOutlineColor = mix(outlineColor2, outlineColor1, outline);
  
  // 添加发光效果
  float glowFactor = outline * glowIntensity;
  finalOutlineColor.rgb += vec3(glowFactor * 0.3, glowFactor * 0.2, 0.0);
  
  finalOutlineColor.a = outline * outlineIntensity;
  gl_FragColor = finalOutlineColor * v_color;
} else {
  // 完全透明
  gl_FragColor = vec4(0.0, 0.0, 0.0, 0.0);
}
}
}%
