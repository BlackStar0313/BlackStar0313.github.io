Shader模拟深海颜色渐变效果

效果​

需求​

  • 钓鱼游戏鱼勾下潜，模拟海洋背景颜色从浅海到深海，颜色渐变
  • 海底总长度2500米，每次深度改变都需要海洋颜色可以动态改变

实现​

可以通过让美术给一张整图，美术来实现整个图片从浅到深的颜色变化，然后滚动整个图片作为背景，但是问题是我们需求是2500米，一米为100像素，也就是这张图需要250000像素，这个大小显然不合适，所以需要使用Shader来实现。

实现分两部分，第一部分是shader实现，

思路为：

  1. 让一张很小的白色纹理，铺满整个屏幕，shader对小纹理进行上下两个颜色进行简单mix，实现uv坐标的由上面浅色到下面深色的颜色渐变。
  2. 美术希望的是比如0到250米是一个浅色到一个深色的渐变，250米到500米是另外一组颜色。而且需要可以根据当前深度变化，所以需要在shader中加入参数供外部改变
  • 深度
  • 顶部当前浅色
  • 顶部目标浅色
  • 底部当前深色
  • 底部目标深色

第二部分是脚本，脚本中根据当前深度，实时传给shader参数来进行颜色渐变。

这样就可以实现目标效果了。

具体操作​

1. 根据UV坐标，按比例对顶部底部颜色进行混合，实现颜色渐变​

OceanGradient.effect
// assets/resources/effect/OceanGradient.effect
// OceanGradient.effect
CCEffect %{
techniques:
- passes:
- vert: vs
  frag: fs
  blendState:
    targets:
    - blend: true
  rasterizerState:
    cullMode: none
  properties:
    texture: { value: white }
    topColor: { value: [0.1, 0.5, 0.9, 1.0], editor: { type: color } }
    bottomColor: { value: [0.0, 0.0, 0.4, 1.0], editor: { type: color } }
    deepTopColor: { value: [0.0, 0.1, 0.5, 1.0], editor: { type: color } }
    deepBottomColor: { value: [0.0, 0.0, 0.2, 1.0], editor: { type: color } }
    currentDepth: { value: 0.0 }
    maxDepth: { value: 2500.0 }
    beforeDepth: { value: 0.0 }
}%

CCProgram vs %{
precision highp float;
#include <cc-global>
#include <cc-local>

in vec3 a_position;
in vec2 a_uv0;
in vec4 a_color;

out vec2 v_uv;
out vec4 v_color;

void main() {
vec4 pos = vec4(a_position, 1);
pos = cc_matViewProj * cc_matWorld * pos;

v_uv = a_uv0;
v_color = a_color;

gl_Position = pos;
}
}%

CCProgram fs %{
precision highp float;
#include <cc-global>

in vec2 v_uv;
in vec4 v_color;

uniform sampler2D texture;
uniform Constant {
vec4 topColor;
vec4 bottomColor;
vec4 deepTopColor;
vec4 deepBottomColor;
float currentDepth;
float maxDepth;
float beforeDepth;
};

void main() {
// 垂直渐变因子，注意我们使用1.0-v_uv.y确保上淡下深
float gradientFactor = 1.0 - v_uv.y;

// 深度比例，用于在表层颜色和深处颜色间插值
float depthFactor = min((currentDepth - beforeDepth) / (maxDepth - beforeDepth), 1.0);

// 根据当前深度，计算当前深度下的顶部颜色
vec4 currentTopColor = mix(topColor, deepTopColor, depthFactor);

// 根据当前深度，计算当前深度下的底部颜色
vec4 currentBottomColor = mix(bottomColor, deepBottomColor, depthFactor);

// 使用当前深度下的顶部和底部颜色，创建渐变效果
vec4 finalColor = mix(currentBottomColor, currentTopColor, gradientFactor);

gl_FragColor = finalColor;
}
}%

4. 添加ts脚本,控制深度及其他参数的更新​

OceanGradient.ts
export default class FishOceanGradientComp extends cc.Component {
@property(cc.Material)
private material: cc.Material = null;
@property(cc.Sprite)
private backgroundSprite: cc.Sprite = null;
@property(cc.Color)
private topColor: cc.Color = new cc.Color(25, 128, 230, 255); // 表层顶部颜色(浅蓝)
@property(cc.Color)
private bottomColor: cc.Color = new cc.Color(0, 0, 102, 255); // 表层底部颜色(深蓝)
@property(cc.Color)
private deepTopColor: cc.Color = new cc.Color(0, 26, 128, 255); // 深处顶部颜色(中蓝)
@property(cc.Color)
private deepBottomColor: cc.Color = new cc.Color(0, 0, 51, 255); // 深处底部颜色(深蓝)
private currentDepth: number = 0;
private currentColorInfo: any = null; // 当前使用的颜色配置

private colorArr: {
	startDepth: number,
	endDepth: number,
	beforeDepth: number,
	topColor: cc.Color,
	bottomColor: cc.Color,
	deepTopColor: cc.Color,
	deepBottomColor: cc.Color,
}[] = [];

start() {
	// 获取材质
	if (!this.material && this.backgroundSprite) {
		this.material = this.backgroundSprite.getMaterial(0);
		console.log("Got material from sprite:", this.material);
	}

	// this.maxDepth = FishingUtils.totalDepth;

	this.initColorArr();

	// 确保材质有效
	if (this.material) {
		console.log("Material is valid, updating parameters");
		this.updateDepth(0);
	} else {
		console.error("No valid material found!");
	}
}

private initColorArr() {
	var c = (color: string) => {
		return cc.color().fromHEX(color);
	}
	var colorArr = [
		{ top: "#95FBF2", bottom: "8AFEF3" },
		{ top: "#65D7FE", bottom: "4BC0FD" },
		{ top: "#0667c1", bottom: "084080" },
		{ top: "#083e7c", bottom: "062f5b" },
		{ top: "#062f59", bottom: "06305A" },
		{ top: "#CFF4DA", bottom: "CFF4DA" },
	]

	this.colorArr = [
		{ startDepth: 0, endDepth: 500, beforeDepth: 0, topColor: c(colorArr[0].top), bottomColor: c(colorArr[0].bottom), deepTopColor: c(colorArr[1].top), deepBottomColor: c(colorArr[1].bottom) },
		{ startDepth: 500, endDepth: 1000, beforeDepth: 500, topColor: c(colorArr[1].top), bottomColor: c(colorArr[1].bottom), deepTopColor: c(colorArr[2].top), deepBottomColor: c(colorArr[2].bottom) },
		{ startDepth: 1000, endDepth: 1500, beforeDepth: 1000, topColor: c(colorArr[2].top), bottomColor: c(colorArr[2].bottom), deepTopColor: c(colorArr[3].top), deepBottomColor: c(colorArr[3].bottom) },
		{ startDepth: 1500, endDepth: 2000, beforeDepth: 1500, topColor: c(colorArr[3].top), bottomColor: c(colorArr[3].bottom), deepTopColor: c(colorArr[4].top), deepBottomColor: c(colorArr[4].bottom) },
		{ startDepth: 2000, endDepth: 2050, beforeDepth: 2000, topColor: c(colorArr[4].top), bottomColor: c(colorArr[4].bottom), deepTopColor: c(colorArr[5].top), deepBottomColor: c(colorArr[5].bottom) },
		{ startDepth: 2050, endDepth: 2500, beforeDepth: 2050, topColor: c(colorArr[5].top), bottomColor: c(colorArr[5].bottom), deepTopColor: c(colorArr[5].top), deepBottomColor: c(colorArr[5].bottom) },
	];
}

/**
 * 更新深度并刷新shader参数
 * @param depth 当前深度（米）
 */
public updateDepth(depth: number) {
	this.currentDepth = depth;
	// 只更新深度参数，减少不必要的调用
	// console.log(`【Ocean】 更新深度: ${this.currentDepth}/${this.maxDepth}`);
	if (this.material) {
		// 找到当前深度对应的颜色配置
		var newColorInfo = this.colorArr.find(x => depth >= x.startDepth && depth <= x.endDepth);

		if (newColorInfo) {
			// 只有当切换到新阶段时才更新颜色配置
			if (!this.currentColorInfo || this.currentColorInfo !== newColorInfo) {
				this.currentColorInfo = newColorInfo;
				this.topColor = newColorInfo.topColor;
				this.bottomColor = newColorInfo.bottomColor;
				this.deepTopColor = newColorInfo.deepTopColor;
				this.deepBottomColor = newColorInfo.deepBottomColor;

				// 更新shader的颜色参数和范围参数
				this.updateShaderParameters();
			}

			// 始终更新当前深度
			this.material.setProperty('currentDepth', this.currentDepth);
		}
	}
}

/**
 * 更新shader的所有参数
 */
private updateShaderParameters() {
	if (!this.material || !this.currentColorInfo) {
		// console.error("No material to update");
		return;
	}

	try {
		// 将颜色转换为0-1范围的数组
		const topColorArray = [
			this.topColor.r / 255,
			this.topColor.g / 255,
			this.topColor.b / 255,
			this.topColor.a / 255
		];

		const bottomColorArray = [
			this.bottomColor.r / 255,
			this.bottomColor.g / 255,
			this.bottomColor.b / 255,
			this.bottomColor.a / 255
		];

		const deepTopColorArray = [
			this.deepTopColor.r / 255,
			this.deepTopColor.g / 255,
			this.deepTopColor.b / 255,
			this.deepTopColor.a / 255
		];

		const deepBottomColorArray = [
			this.deepBottomColor.r / 255,
			this.deepBottomColor.g / 255,
			this.deepBottomColor.b / 255,
			this.deepBottomColor.a / 255
		];

		// 设置颜色参数
		this.material.setProperty('topColor', topColorArray);
		this.material.setProperty('bottomColor', bottomColorArray);
		this.material.setProperty('deepTopColor', deepTopColorArray);
		this.material.setProperty('deepBottomColor', deepBottomColorArray);

		// 设置深度范围参数
		this.material.setProperty('beforeDepth', this.currentColorInfo.beforeDepth);
		this.material.setProperty('maxDepth', this.currentColorInfo.endDepth);
		this.material.setProperty('currentDepth', this.currentDepth);


		console.log("Shader parameters updated successfully");
	} catch (error) {
		console.error("Error updating shader parameters:", error);
	}
}
}
